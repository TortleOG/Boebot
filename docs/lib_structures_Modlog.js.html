/**
 * Creates an embed for moderation actions.
 */
class Modlog {
  /**
   * @param {Boebot} client The client the bot was created with.
   * @param {string} type The type of moderation action.
   */
  constructor(client, type) {
    /**
     * BoeBot Client
     * @type {BoeBot}
     */
    Object.defineProperty(this, "client", { value: client });

    /**
     * The Embed type
     * @type {string}
     */
    this.type = type;

    if (!this.title) throw `Unknown action (${this.type}) found.`;
  }

  /**
   * Sets the offending member for the embed.
   * @param {Object} member The offending member.
   * @return {Function|*} createEmbed function if all properties are set, or this.
   */
  setMember(member) {
    if (!member) throw "Expected input 'member' but none was found.";
    this.member = {
      name: this.type === "unban" ? member.tag : member.user.tag,
      id: member.id,
    };
    if (this.member && this.moderator && this.reason) return this.createEmbed();
    return this;
  }

  /**
   * Sets the moderator for the embed.
   * @param {Object} mod The moderator that created the action.
   * @return {Function|*} createEmbed function if all properties are set, or this.
   */
  setMod(mod) {
    if (!mod) throw "Expected input 'mod' but none was found.";
    this.moderator = {
      name: mod.user.tag,
      id: mod.id,
    };
    if (this.member && this.moderator && this.reason) return this.createEmbed();
    return this;
  }

  /**
   * Sets the reason in the embed.
   * @param {string} reason The reason for the action.
   * @return {Function|*} createEmbed function if all properties are set, or this.
   */
  setReason(reason) {
    if (!reason) throw "Expected input 'reason' but none was found.";
    if (typeof reason !== "string") throw "Expected input 'reason' to be a string.";
    this.reason = reason;
    if (this.member && this.moderator && this.reason) return this.createEmbed();
    return this;
  }

  /**
   * The embed title.
   * @type {string}
   */
  get title() {
    const past = {
      ban: "Banned",
      unban: "Unbanned",
      mute: "Muted",
      unmute: "Unmuted",
      kick: "Kicked",
      warn: "Warned",
    };
    return past[this.type];
  }

  /**
   * The hex color for the embed.
   * @type {number}
   */
  get color() {
    const embedTypes = {
      ban: 0xcc0000,
      unban: 0x2d862d,
      kick: 0xe65c00,
      mute: 0x993366,
      unmute: 0x993366,
      warn: 0xf2f20d,
    };
    return embedTypes[this.type];
  }

  /**
   * Creates the embed
   * @return {external:Discord.MessageEmbed} The MessageEmbed
   */
  createEmbed() {
    const embed = new this.client.methods.Embed()
      .setTitle(`User ${this.title}`)
      .setColor(this.color)
      .setDescription([
        `**Member**: ${this.member.name} | ${this.member.id}`,
        `**Moderator**: ${this.moderator.name}`,
        `**Reason**: ${this.reason}`,
      ].join("\n"))
      .setTimestamp()
      .setFooter(this.type.capitalize(), this.client.user.displayAvatarURL({ format: "jpg" }));
    return embed;
  }
}


module.exports = Modlog;
